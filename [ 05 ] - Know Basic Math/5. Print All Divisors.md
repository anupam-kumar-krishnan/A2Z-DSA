# _Print All Divisors_

```cpp
#include<bits/stdc++.h>
using namespace std;
int main()
{
  int n;
  cin>>n;

  for(int i=1;i<=n;i++){
    if(n%i==0)
      cout<<i<<" ";
  }    
}

// Time Complexity => O(n)
```

**_We can do it in a better Time Complexity_**

```cpp
#include<bits/stdc++.h>
using namespace std;
int main()
{
  int n;
  cin>>n;

  vector<int> ls;
  
  // Instead of using sqrt
  // i * i <= n
  // 6 * 6 <= 36
  // 7 * 7 <=36
  for(int i=1;i<=sqrt(n);i++){
    if(n%i==0){
      ls.push_back(i);
      if((n/i)!=i){
         ls.push_back(n/i);
      }
    }
  }
  

 // O(no of factors * log(no of factors)): n is the number of factors
  sort(ls.begin(),ls.end());

// O(number of factors)
  for(auto it: ls)
    cout<<it<< " ";
}

// TC => O(sqrt(n))
```
