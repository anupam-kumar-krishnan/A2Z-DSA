# _Prime Number_

```cpp
#include<bits/stdc++.h>
using namespace std;
int main()
{
  int n,cnt=0;
  cin>>n;

  for(int i=1;i<=n;i++){
    if(n%i==0){
      cnt++;
    }
  }
  if(cnt==2)
    cout<<"Prime Number";
  else
    cout<<"Not a Prime Number";
}

// Extreme Brute Force Approach
// TC => O(N)
```

### _Optimal Approach_

```cpp
#include<bits/stdc++.h>
using namespace std;
int main()
{
  int n,cnt=0;
  cin>>n;

  // sqrt is alo written in this way
  // i * i <= n
  for(int i=1;i*i<=n;i++){
    if(n%i==0){
      cnt++;
      if((n/i)!=i){
        cnt++;
      }
    }
  }
  if(cnt==2)
    cout<<"Prime Number";
  else
    cout<<"Not a Prime Number";
}

// Optimal Approarch
// TC => O(sqrt(N))
```
